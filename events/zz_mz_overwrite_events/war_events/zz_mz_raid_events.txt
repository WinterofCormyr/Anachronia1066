raiding.0021 = { # Increase raid tally
	hidden = yes
	scope = army

	trigger = {
		exists = army_commander
	}

	immediate = {
		root.army_commander = {
			if = {
				limit = {
					NOT = { has_variable = raid_counter }
				}
				set_variable = {
					name = raid_counter
					value = 1
				}
			}
			else = {
				change_variable = {
					name = raid_counter
					add = 1
				}
			}
			if = { # If viking is virtuous, 50% chance to add extra raid talley
				limit = {
					faith = { trait_is_virtue = viking }
				}
				random = {
					chance = 50
					change_variable = {
						name = raid_counter
						add = 1
					}
				}
			}	
		}
	}
}

raiding.0022 = { # Chance to fire viking/raider trait acquisition event
	hidden = yes
	scope = army

	trigger = {
		exists = army_commander
		army_commander = {
			has_variable = raid_counter
			var:raid_counter >= 20
			NOT = { has_trait = viking }
		}
	}

	immediate = {
		root.army_commander = { save_scope_as = raid_leader }
		scope:raid_leader = {
			random = {
				chance = 20
				trigger_event = {
					id = raiding.0023
					days = { 180 545 }
				}
				modifier = { # Extra 10% chance if viking is virtuous
					add = 10
					faith = { trait_is_virtue = viking }
				}
				modifier = {
					add = 10
					var:raid_counter >= 30
				}
				modifier = {
					add = 10
					var:raid_counter >= 40
				}
				modifier = {
					add = 10
					var:raid_counter >= 50
				}
			}
		}
	}
}
